function onFormSubmit(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Events');
  var responses = e.values;
  
  // Assuming form responses are in order:
  // [Timestamp, ActionType, EventID/Date, Name/TimeStart, TimeEnd, Location, Organizer]
  
  var actionType = responses[1]; // Action Type
  
  if (actionType === 'join') {
    // Join existing event
    var eventId = responses[2];
    var name = responses[3];
    
    // Find the row with matching ID
    var data = sheet.getDataRange().getValues();
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] == eventId) { // Column A is ID
        // Append name to Participants (Column E)
        var currentParticipants = data[i][4] || '';
        var newParticipants = currentParticipants 
          ? currentParticipants + ', ' + name 
          : name;
        sheet.getRange(i + 1, 5).setValue(newParticipants);
        break;
      }
    }
  } else if (actionType === 'create') {
    // Create new event
    var date = responses[2];
    var timeStart = responses[3];
    var timeEnd = responses[4];
    var location = responses[5];
    var organizer = responses[6];
    
    // Generate new ID
    var lastRow = sheet.getLastRow();
    var newId = lastRow > 1 ? sheet.getRange(lastRow, 1).getValue() + 1 : 1;
    
    // Combine date and time
    var dateTime = Utilities.formatDate(new Date(date), Session.getScriptTimeZone(), 'yyyy-MM-dd') + 'T' + timeStart;
    
    // Add new row
    sheet.appendRow([
      newId,
      dateTime,
      location,
      organizer,
      organizer, // Organizer is first participant
      'active'
    ]);
  }
}
